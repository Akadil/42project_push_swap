int main(int argc, char **argv)
{
    if (ft_check(argc, argv) == 0)
        return (ft_error());
    str = get_next_line(0);
    while (!str)
    {
        ft_do_command(a, b, str);
        str = get_next_line(0);
    }
    if (ft_is_sorted(a) == 1 and ft_is_empty(b) == 1)
        write(1, "Ok\n", 3);
    else
        write(1, "Ko\n", 3);
    return (0);
}
